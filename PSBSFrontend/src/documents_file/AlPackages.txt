CREATE TABLE Packages (
    id INT IDENTITY(1,1) PRIMARY KEY,

    package_name NVARCHAR(100) NOT NULL,
    description NVARCHAR(MAX) NOT NULL,

    coverage_duration_hours INT NOT NULL,
    max_edited_photos INT NOT NULL,

    raw_files_available BIT NOT NULL DEFAULT 0,

    base_price DECIMAL(10,2) NOT NULL,

    created_at DATETIME2 NOT NULL DEFAULT GETDATE(),
    updated_at DATETIME2 NOT NULL DEFAULT GETDATE()
);

CREATE TABLE AddOns (
    id INT IDENTITY(1,1) PRIMARY KEY,

    addon_name NVARCHAR(100) NOT NULL,
    addon_description NVARCHAR(255),
    addon_price DECIMAL(10,2) NOT NULL,

    created_at DATETIME2 NOT NULL DEFAULT GETDATE()
);



CREATE TABLE PackageAddOns (
    id INT IDENTITY(1,1) PRIMARY KEY,

    package_id INT NOT NULL,
    addon_id INT NOT NULL,

    FOREIGN KEY (package_id) REFERENCES Packages(id) ON DELETE CASCADE,
    FOREIGN KEY (addon_id) REFERENCES AddOns(id) ON DELETE CASCADE,

    CONSTRAINT UQ_Package_AddOn UNIQUE (package_id, addon_id)
);
